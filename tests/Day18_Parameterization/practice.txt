// multi dimentional data
import {test,expect} from "@playwright/test"


const testdata:string[][]=[
    ["joysondsouza@gmail.com","joys1234","valid","attempt_1"],
    ["joysondsouza@gmail.com","joy123","invalid","attempt_2"]
]

for(const [email,password,validation,attempt] of testdata){

    test(`testdata ${attempt}`,async({page})=>{

    await page.goto("URL");
    await page.locator("#email").fill(email)
    await page.locator("#password").fill(password)
    await page.locator("#login").click();

    if(validation==="valid")
    {
          await expect(page.locator("logoutpage")).toBeVisible();
    }
    else
    {
          await expect(page.locator("loginpage")).toBeVisible();
    }
    
})
}
